package group.project;

import java.util.ArrayList;
import java.util.Random;

/**
 *
 * @author Evan Schwarzenbach
 */
public class User {
    
    private String Name;
    private String Pass;
    private String Rank;
    private String id;
    public User(String name, String pass, String rank){
        Name=name;
        Pass=pass;
        Rank=rank;
        Random rand= new Random();
        id= String.format("%04d", rand.nextInt(10000)); 
        
    }
    
    public String getName(){
        return Name;
        
    }
    
    public String getPass(){
        return Pass;
        
    }
    public String getRank(){
        return Rank;
        
    }
    public String getID(){
        return id;
        
    }

    public String[] getInfo(){
        String ar[] = new String[4];
        ar[0]= Name;
        ar[1]= Pass;
        ar[2]= Rank;
        ar[3]= id;
        return ar; 

    }
}
