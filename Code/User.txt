
package group.project;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Random;

/**
 *
 * @author Evan Schwarzenbach
 */
public class User {
    private ArrayList<Course> myCourses = new ArrayList<>();
    private String Name;
    private String Pass;
    private String Rank;
    private String id;
    public User(String name, String pass, String rank){
        Name=name;
        Pass=pass;
        Rank=rank;
        Random rand= new Random();
        id= String.format("%05d", rand.nextInt(100000)); 
        
    }
    
    public String getName(){
        return Name;
        
    }
    
    public String getPass(){
        return Pass;
        
    }
    public String getRank(){
        return Rank;
        
    }
    public String getID(){
        return id;
        
    }

    public String[] getInfo(){// Used for printing all users
        String ar[] = new String[4];
        ar[0]= Name;
        ar[1]= Pass;
        ar[2]= Rank;
        ar[3]= id;
        return ar; 

    }
    
    public void addCourse(Course course) {// Each user will have their own list of courses they are in
        myCourses.add(course);
    }
    
    public void myCourses() {
        if (myCourses.isEmpty()){
            System.out.println("You have no courses");
        }
        else{
            System.out.println("Your courses are:");
            for (int i =0; i < myCourses.size(); i++){
            System.out.println(Arrays.toString(myCourses.get(i).getInfo()));
            }
        }

    }

}
